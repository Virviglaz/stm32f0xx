98static inline uint32_t get_pll_value(void)
99{
100 return ((RCC->CFGR & RCC_CFGR_PLLMUL) >> 18) + 2;
101}
102
108static inline void set_pll_value(uint32_t value)
109{
110 RCC->CFGR &= ~RCC_CFGR_PLLMUL;
111 RCC->CFGR |= (value - 2) << 18;
112}
113
118static inline uint32_t get_pll_prediv(void)
119{
120 return (RCC->CFGR2 & RCC_CFGR2_PREDIV1) + 1;
121}
122
128static inline void set_pll_prediv(uint32_t value)
129{
130 RCC->CFGR2 &= ~RCC_CFGR2_PREDIV1;
131 RCC->CFGR2 |= value - 1;
132}
133
139static inline void set_pll_source(enum pll_input_clock_src_t src)
140{
141 if (src == HSE_PREDIV)
142 RCC->CFGR |= RCC_CFGR_PLLSRC_1;
143 else
144 RCC->CFGR &= ~RCC_CFGR_PLLSRC_1;
145}
146
151static inline enum pll_input_clock_src_t get_pll_source(void)
152{
153 return RCC->CFGR & RCC_CFGR_PLLSRC_1 ? HSE_PREDIV : HSI_DIV2;
154}

161static inline void set_ahb_clock_source(enum ahb_clock_source_t src)
162{
163 RCC->CFGR &= ~RCC_CFGR_SW;
164 RCC->CFGR |= src;
165}
166
171static inline enum ahb_clock_source_t get_ahb_clock_source(void)
172{
173 return (enum ahb_clock_source_t)((RCC->CFGR & RCC_CFGR_SWS) >> 2);
174}
175
181static inline void set_ahb_clock_divider(enum ahb_clock_divider_t div)
182{
183 RCC->CFGR &= ~RCC_CFGR_HPRE;
184 RCC->CFGR |= div << 4;
185}
186
191static inline enum ahb_clock_divider_t get_ahb_clock_divider(void)
192{
193 uint32_t ret = (RCC->CFGR & RCC_CFGR_HPRE) >> 4;
194 return (enum ahb_clock_divider_t)(ret < AHB_DIV_1 ? AHB_DIV_1 : ret);
195}
196
202static inline void set_apb_clock_divider(enum apb_clock_divider_t div)
203{
204 RCC->CFGR &= ~RCC_CFGR_PPRE;
205 RCC->CFGR |= div << 8;
206}
207
212static inline enum apb_clock_divider_t get_apb_clock_divider(void)
213{
214 uint32_t ret = (RCC->CFGR & RCC_CFGR_PPRE) >> 8;
215 return (enum apb_clock_divider_t)(ret < APB_DIV_1 ? APB_DIV_1 : ret);
216}
