67/* define i2c message operation */
68enum i2c_dir {
69 I2C_WRITING = 0,
70 I2C_READING,
71};

73/* i2c message is used to define single transfer i2c action */
74struct i2c_msg_t {
75 enum i2c_dir dir;
76 uint8_t *data;
77 uint8_t size;
78};
79
80/* transfer handles the messages */
81struct i2c_xfer {
82 uint8_t address;
83 struct i2c_msg_t *msgs;
84 uint8_t msgs_left;
85 isr_handler_t handler;
86 void *prv_ptr;
87 bool done;
88 uint16_t cnt; /* byte counter for isr mode */
89 int error;
90};

92typedef struct i2c_dev {
93 I2C_TypeDef *i2c;
94 struct i2c_xfer xfer;
95#ifdef FREERTOS
96 SemaphoreHandle_t done;
97 SemaphoreHandle_t lock;
98 int err;
99#endif
100} i2c_dev_t;
