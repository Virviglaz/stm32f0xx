69struct uart_buf_t {
70 struct {
71 char *buf; /* data will be stored here */
72 char stop; /* end of data marker */
73 uint16_t size; /* maximum buffer size */
74 uint16_t cnt; /* byte counter */
75 uart_handler_t handler; /* buffer full/new line handler */
76 void *data; /* private data for generic use */
77 uint32_t error; /* error flags will be stored here */
78 uart_rx_single_byte_t s_byte_rcv; /* single byte rcv handler */
79 volatile bool done; /* set to true when done */
80#ifdef FREERTOS
81 SemaphoreHandle_t slock;
82 SemaphoreHandle_t sdone;
83 } rx;
84#endif
85 struct {
86 char *buf; /* data will be taken from here */
87 uint16_t size; /* amount of bytes to transfer */
88 uint16_t cnt; /* byte counter */
89 uart_handler_t handler; /* end of transfer handler */
90 void *data; /* private data for generic use */
91 volatile bool done; /* set to true when done */
92#ifdef FREERTOS
93 SemaphoreHandle_t slock;
94 SemaphoreHandle_t sdone;
95 } tx;
96#endif
97};
